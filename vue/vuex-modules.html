# Vuex 模块化实践

Vuex 直接把所有东西写在 store.js里是可以的，但是这样一个文件会非常混乱，且不同业务逻辑之间需要有模块的划分 所以vuex模块化是普遍需求：

正好这次在 nuxt 项目中需要用到vuex的管理全局state，记录一下学习过程：

# store.js

import Vue from 'vue';
import Vuex from 'vuex';

import iconfont from './icons' // 引入一个模块 icons

Vue.use(Vuex);

const store = () => new Vuex.Store({

  state: { // 全局state, 无命名空间
    counter: 0
  },
  mutations: {  // 全局mutation, 无命名空间
    increment(state) {
      state.counter++;
    }
  },
  modules: {  // 注册模块
    iconfont  // 缩写，  iconfont: iconfont
  }
});

export default store;

# icon.js (module)

// module: iconfont
const state = {  // 模块下的state
  hasLoaded: false,
};

const getters = {  // 模块下的getters
  loadedState: state => {
    return !state.hasLoaded;
  }
};

const mutations = {  // 模块下的mutation
  ICONS_LOADED(state) {
    state.hasLoaded = true;
  }
};

const actions = {  // 模块下的 actions
  hasSetIconFontScriptTag({state, commit}) {
    commit('ICONS_LOADED')
  }
};

export default {
  namespaced: true,  // 启用命名空间
  state,
  getters,
  mutations,
  actions
}

# 使用：（Icon.vue）

<template>
  <svg class="icons"
  >
    <use :xlink:href="'#i-'+type"/>
  </svg>
</template>

<script>
  import {mapState, mapMutations, mapActions, mapGetters} from 'vuex';

  export default {
    props: {
      type: {
        type: String,
        default: 'home'
      }
    },
    computed: {
      ...mapGetters('iconfont', ['loadedState'])
    },
    created() {
      // map系列函数 帮你从指定的命名空间里取出相应的字段，合并成一个obj，使用起来更方便
      console.log('mapState:', mapState('iconfont', ['hasLoaded']));
      console.log('mapGetters:', mapGetters('iconfont', ['loadedState']));
      console.log('mapMutations:', mapMutations('iconfont', ['ICONS_LOADED']));
      console.log('mapActions:', mapActions('iconfont', ['hasSetIconFontScriptTag']));
    }
  };
</script>

<style scoped>
  .icons {
    width: 1em;
    height: 1em;
    vertical-align: -0.15em;
    fill: currentColor; /* fill用currentColor则会自动匹配当前的color属性值 */
    overflow: hidden;
  }
</style>

  </section>

<script>
  import { mapState, mapMutations, mapActions, mapGetters } from 'vuex';

  },
  methods: {
    ...mapMutations('filter', [

    ]),

    ]),

    }
  },
  mounted() {
  },
  beforeDestroy() {
  }
};
</script>
