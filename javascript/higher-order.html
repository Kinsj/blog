# call bind apply理解

# 先写结果：

  • 这三个函数都是高阶函数
  • 用于对函数的高级操作，比如 绑定this、封装参数等

# call

call是最好理解，也是其他两个方法的基础

var f1 = function() {
  console.log("this", this);
  console.log("arguments", arguments);
}

f1.call({name: 'kin'}, 1,2,3)
// this: {name: 'kin'}
// arguments: [1,2,3]

可以得出结论：
f1.call(this, arguments)
给f1绑定了this,并传入剩余参数，并调用这个函数

# bind

var bind = Function.prototype.bind
var f1 = function() {
  console.log("this", this);
  console.log("arguments", arguments);
}
f1.bind({name: 'kin'}, 1,2,3)()  // 不会执行

// 要搞懂bind，就必须搞懂 bind.call
// obj.method(a,b,c,d) 等价于
// obj.method.call(obj, a,b,c,d)
// 设 obj = f1, method = bind
// f1.bind(a,b,c,d) 等价于
// f1.bind.call(f1, a,b,c,d)
// 带入 参数： a = {name: 'kin'} b,c,d = 1,2,3
// f1.bind({name: 'kin'}, 1, 2, 3) 等价于
// f1.bind.call(f1, {name: 'kin'}, 1, 2, 3)
// 因为 f1.bind === Function.prototype.bind === bind
// 所以以上等于 bind.call(f1, {name: 'kin'}, 1, 2, 3)
// 得出以下结论：
bind.call(f1, {name: 'kin'}, 1,2,3)

bind.call 接受 一个函数， 一个this， 其他参数
所以 f1.bind(this, argment)
给f1绑定了this,并传入剩余参数，并将这个函数返回

# apply

var apply = Function.prototype.bind
var f1 = function() {
  console.log("this", this);
  console.log("arguments", arguments);
}
f1.apply({name: 'kin'}, [1,2,3])

// 要搞懂apply，就必须搞懂 apply.call
// obj.method(a,b,c,d) 等价于
// obj.method.call(obj, a,b,c,d)
// 设 obj = f1, method = apply
// f1.apply(a,b) 等价于
// f1.apply.call(f1, a,b)
// 带入 参数： a = {name: 'kin'} b = [1,2,3]
// f1.apply({name: 'kin'}, [1,2,3]) 等价于
// f1.apply.call(f1, {name: 'kin'}, [1,2,3])
// 因为 f1.apply === Function.prototype.apply === apply
// 所以以上等于 apply.call(f1, {name: 'kin'}, [1,2,3])
// 得出以下结论：
apply.call(f1, {name: 'kin'}, [1,2,3])

apply.call 接受 一个函数， 一个this， 其他参数组成的数组
所以 f1.apply(this, array)
给f1绑定了this,并传入剩余参数组成的数组，并调用这个函数
