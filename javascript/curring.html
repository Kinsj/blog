# 柯里化

柯里化主要就是把多参数的函数转换成但参数函数，单参数函数的好处在函数式中有所体现，这里不做细化

# 例1

如何把三参数函数 add(1,2,3) 变成：
curriedAdd(1)(2)(3)形式？

const curriedAdd = 
    a => b => c => add(a,b,c) // 就这么简单

# 例2

currify(addTwo)(1)(2) // 3
currify(addThree)(1)(2)(3) // 6
currify(addFour)(1)(2)(3)(4) // 10
实现这个currify

// 基本实现
function addThree(a,b,c) {
  return a+b+c;
}

const currify = (fn, params = []) =>{
  return (arg) => {
    params.push(arg)
    if(fn.length === params.length) {
      return fn(...params)
    } else {
      return currify(fn, [...params])
    }
  }
}
currify(addThree)(1)(2)(3); // 6

// 优化： 支持 currify(addFour)(1)(2,3)(4) 这种形式
const currify = (fn, params = []) =>{
  return (...args) => {
    if(fn.length === params.length + args.length) {
      return fn(...params, ...args)
    } else {
      return currify(fn, [...params, ...args])
    }
  }
}

// 简化代码：
const currify = (fn, params = []) => 
  (...args) => 
    fn.length === params.length + args.length
      ? fn(...params, ...args)
      : currify(fn, [...params, ...args])
