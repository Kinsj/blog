# 测试用例（测试驱动TDD）

const bind2 = require('./bind');

test1('可以绑定this')
test2('可以在绑定时传入参数')
test3('可以在绑定和调用时传入参数')
test4('可以在绑定new')
test5('绑定后可以拥有原型')

function test1(msg) {
  Function.prototype.bind2 = bind2
  console.log(msg);
  function fn() {
    return this
  }
  const newFn = fn.bind2({name: 'kin'})
  console.assert(newFn().name === 'kin')
}

function test2(msg) {
  Function.prototype.bind2 = bind2
  console.log(msg);
  function fn(a, b) {
    return [this, a, b]
  }
  const newFn = fn.bind2({name: 'kin'}, 1, 2)
  console.assert(newFn()[0].name === 'kin', 'test2: this')
  console.assert(newFn()[1] === 1, 'test2: a')
  console.assert(newFn()[2] === 2, 'test2: b')
}

function test3(msg) {
  console.log(msg);
  Function.prototype.bind2 = bind2

  function fn(a, b) {
    return [this, a, b]
  }

  const newFn = fn.bind2({name: 'kin'}, 1)
  console.assert(newFn(2)[0].name === 'kin', 'test3: this')
  console.assert(newFn(2)[1] === 1, 'test3: a')
  console.assert(newFn(2)[2] === 2, 'test3: b')
}

function test4(msg) {
  console.log(msg)
  Function.prototype.bind2 = bind2

  function fn(a) {
    this.a = a
  }

  const fn2 = fn.bind2(undefined, 'x');
  console.assert(new fn2().a === 'x', 'test4: new')
}

function test5(msg) {
  console.log(msg)
  Function.prototype.bind2 = bind2

  function fn(a) {
    this.a = a
  }
  fn.prototype.sayHi = function() {}

  const fn2 = fn.bind2(undefined, 'x');
  console.assert(new fn2().a === 'x', 'test5: new')
  console.assert(new fn2().sayHi !== undefined, 'test5: prototype')
}

// 简易版 用了 es6 语法
const _bind2 = function(asThis, ...args) {
  var fn = this
  function resultFn (...args2) {
    const isNew = this instanceof resultFn;
    return fn.call(isNew ? this : asThis, ...args, ...args2)
  }
  resultFn.prototype = fn.prototype
  return resultFn
}

// 不用高级语法版： （没有bind，一般也没有其他高级语法
var slice = Array.prototype.slice
const bind2 = function() {
  var fn = this
  var asThis = arguments[0]  // 取出参数中的第一项作为this
  var restArgs = slice.call(arguments, 1) // slice.call可以接受arguments作为参数
  function resultFn() {
    var isNew = this instanceof resultFn;
    return fn.apply(
      isNew ? this : asThis, 
      restArgs.concat(slice.call(arguments)) // 这里必须用slice.call把arguments转为数组
    )
  }
  resultFn.prototype = fn.prototype
  return resultFn
}

module.exports = bind2
